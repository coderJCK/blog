# 条件语句

# if

// 1、单分支 if
if(条件){
  // 条件为true时执行的代码
}
// 2、双分支 if else
 
if(条件){
  // 条件为true时执行的代码
}else{
  // 条件为true时执行的代码
}
// 3、多条 if else if else
if(条件1){
    // 条件1为true时执行的代码
}else if(条件2){
    // 条件1为false,条件2为true时执行的代码
}else{
    // 条件1、2都为false时执行的代码
}

# switch

switch(条件){
    case 值1:
        // 条件为值1时执行的代码块
        break；
    case 值2:
        // 条件为值2时执行的代码块
        break；
    case 值3:
        // 条件为值3时执行的代码块
        break；
    default:
        // 条件为其他时执行的代码块
        break;
}

# 循环语句

# while

while(条件表达式){
	// 代码块/循环体；
}
 
// while循环计算1+...+100
var num = 0;
var i = 1;
while (i <= 100) {
  num += i;
  i++;
}

# do while

  num += i;
  i++;

# for

// 语句1循环开始前执行
// 语句2运行循环的条件
// 语句3循环执行后执行
for (语句 1; 语句 2; 语句 3){
    // 被执行的代码块
}

// 如计算1+...+100
var num = 0;
for(var i=1;i<=100;i++){
    num += i;
}

# for in

循环对象的键名和键值，即属性名和属性值

var obj = {fname:"Bill",lname:"Gates",age:56}; 
for(x in obj){
    console.log(x, obj[x])
    /**
    * fname Bill
    * lname Gates
    * age 56
    */
}
